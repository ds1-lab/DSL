def selection(salaries):
    n=len(salaries)

    for i in range(n):
        min_idx=i

        for j in range (i+1 , n):
            if salaries[j]<salaries[i]:
                min_idx=j
                
                salaries[i],salaries[min_idx]=salaries[min_idx],salaries[i]

salaries=[]

while True:
    print("\n===Bubble Sort===")
    print("1  Enter Salaries")
    print("2  Sort in assending order (Selection Sort)")
    print("3  Show top 5 Salaries")
    print("4  Exit")

    ch = int(input("\nEnter Your Choice : "))

    if ch == 1:
        Emp=int(input("\nEnter No. of Employees: "))
        for i in range(Emp):
            sal_inp=int(input(f"\nEnter Salary For Employee no. {i+1} : "))
            salaries.append(sal_inp)

    elif ch == 2:
        selection(salaries)
        print("\nSalaries In Sorted Order (Selection Sort): ")
        print(salaries)

    elif ch == 3:
        top_5=salaries[-5:]
        top_5.reverse()
        for i in top_5:
            print(i)

    elif ch== 4:
        print("Exiting...")
        break

    else:
        print("Invalid Input...")
