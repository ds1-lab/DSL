from collections import deque

graph={}

n=int(input("\nEnter No. Of Locations :  "))
for i in range (n):
    graph[i] = []

e=int(input("\nEnter No. Of Routes :  "))
for i in range (e):
    u=int(input("\nFrom : "))
    v=int(input("To : "))
    graph[u].append(v)
    graph[v].append(u)

print("\nAdjecency Matrix :")
for i in graph:
    print(f"{i}  {graph[i]}")

start=int(input("\nEnter Start point : "))

visited= []
queue=deque([start])

while queue:
    node=queue.popleft()
    if node not in visited:
        visited.append(node)
        for n in graph[node]:
            if n not in visited:
                queue.append(n)

print("\nBFS Sequence:")
print(visited)
