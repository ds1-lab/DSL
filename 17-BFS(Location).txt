from collections import deque

def bfs(graph, start):
    visited = []
    queue = deque([start])
    while queue:
        node = queue.popleft()
        if node not in visited:
            visited.append(node)
            for neighbour in graph[node]:
                if neighbour not in visited:
                    queue.append(neighbour)
    return visited

graph = {}
n = int(input("Enter number of locations: "))
for i in range(n):
    graph[i] = []
e = int(input("Enter number of routes (connections): "))
print("Enter connections (example: 0 1 means route between 0 and 1):")
for _ in range(e):
    u, v = map(int, input().split())
    graph[u].append(v)
    graph[v].append(u)

print("\nAdjacency List of the Area:")
for key, value in graph.items():
    print(key, ":", value)

start = int(input("\nEnter starting location: "))
order = bfs(graph, start)
print("\nSequence of locations visited using BFS:")
print(order)
