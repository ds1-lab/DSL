class undo_redo:

    def __init__(self):
        self.undo_stack = []
        self.redo_stack = []
        self.doc= ""

    def edit(self,new_txt):

        self.doc=new_text
        self.redo_stack.clear()

    def undo(self):
        if not self.undo_stack:
            print("Nothing to Undo")
            return
        
        self.redo_stack.append(self.doc)
        self.doc=self.undo_stack.pop()
        print("Undone Successfully")
        system.Display()

    def redo(self):
        if not self.redo_stack:
            print("Nothing to redo")
            return
        
        self.undo_stack.append(self.doc)
        self.doc=self.redo_stack.pop()
        print("Redone Successfully")
        system.Display()

    def Display(self):
        print("\n")
        print(self.doc if self.doc else "\nNothing to Display")
