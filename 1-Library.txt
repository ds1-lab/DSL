from collections import Counter

print("Library Borrow Records System")

books = []
borrows = []

# Taking input
n = int(input("\nEnter number of books: "))
for i in range(n):
    name = input(f"Enter book name: ")
    count = int(input(f"Enter number of times '{name}' was borrowed: "))
    books.append(name)
    borrows.append(count)

# Menu
while True:
    print("\n===== MENU =====")
    print("1. Average number of borrowings")
    print("2. Book with highest borrowings")
    print("3. Book with lowest borrowings")
    print("4. Books never borrowed (0 times)")
    print("5. Most frequently borrowed count (Mode)")
    print("6. Exit")

    ch = int(input("Enter your choice: "))

    if ch == 1:
        avg = sum(borrows) / len(borrows)
        print(f"Average borrowings: {avg:.2f}")

    elif ch == 2:
        high = max(borrows)
        print("Book with highest borrowings:", books[borrows.index(high)], "→", high)

    elif ch == 3:
        low = min(borrows)
        print("Book with lowest borrowings:", books[borrows.index(low)], "→", low)

    elif ch == 4:
        for i in range (len(books)):
            zero_count=borrows.count(0)
        print("Books never borrowed (0 times):",zero_count)

    elif ch == 5:
        freq = Counter(borrows)
        mode = freq.most_common(1)[0][0]
        print("Most frequently occurring borrow count (mode):", mode)

    elif ch == 6:
        print("Exiting... Thank you!")
        break

    else:
        print("Invalid choice, try again.")
