def selection(Percentage):
    n=len(Percentage)

    for i in range(n):
        min_idx=i

        for j in range (i+1 , n):
            if Percentage[j]<Percentage[i]:
                min_idx=j
                
                Percentage[i],Percentage[min_idx]=Percentage[min_idx],Percentage[i]

Percentage=[]

while True:
    print("\n===Bubble Sort===")
    print("1  Enter Percentage")
    print("2  Sort in assending order (Selection Sort)")
    print("3  Show top 5 Percentage")
    print("4  Exit")

    ch = int(input("\nEnter Your Choice : "))

    if ch == 1:
        Emp=int(input("\nEnter No. of Students: "))
        for i in range(Emp):
            sal_inp=int(input(f"\nEnter Salary For Student no. {i+1} : "))
            Percentage.append(sal_inp)

    elif ch == 2:
        selection(Percentage)
        print("\nPercentage In Sorted Order (Selection Sort): ")
        print(Percentage)

    elif ch == 3:
        top_5=Percentage[-5:]
        top_5.reverse()
        for i in top_5:
            print(i)

    elif ch== 4:
        print("Exiting...")
        break

    else:
        print("Invalid Input...")
